/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val   int
 *     Left  *TreeNode
 *     Right *TreeNode
 * }
 */

func lowestCommonAncestor(root, p, q *TreeNode) *TreeNode {
	lv := p.Val
    rv := q.Val
    if lv > rv {
        lv, rv = rv, lv
    }

    // 递归
    // if root.Val < lv {

    // } else if root.Val > rv {
    //     return lowestCommonAncestor(root.Left, p, q)
    // } else {
    //     return root
    // }

    cur := root
    for cur != nil {
        if cur.Val < lv {
            cur = cur.Right
        } else if cur.Val > rv {
            cur = cur.Left
        } else {
            return cur
        }
    }
    return nil
}
