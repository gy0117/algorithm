## 快速排序

思想：分治

1. 确定分界点，一般可以取左端点，中间节点，右端点，或者随机
2. 调整区间：确定一个x元素的位置，使得其左的元素都小于等于x，其右边的元素都大于等于x
3. 递归处理左右两段

```go
func quickSort(nums []int, start, end int) {
    if start >= end {
        return
    }
    j := partition(nums, start, end)
    quickSort(nums, start, j)
    quickSort(nums, j+1, end)
}

// 取中间节点比较好，如果取左端点，当数据量大的时候可能有问题
// 取中间节点，最后退出循环时，不再需要交换nums[j]和x的位置
func partition(nums []int, start, end int) int {
    x := nums[start + (end - start) / 2]
    i, j := start - 1, end + 1
    
    for i < j {
        for i < end {
            i++
            if nums[i] >= x {
                break
            }
        }
        for j > start {
            j--
            if nums[j] <= x {
                break
            }
        }
        if i >= j {
            break
        }
        nums[i], nums[j] = nums[j], nums[i]
    }
    return j
}

// 如果是取左端点
func partition(nums []int, start, end int) int {
    x := nums[start]
    i, j := start, end + 1
    
    for i < j {
        for i < end {
            i++
            if nums[i] >= x {
                break
            }
        }
        for j > start {
            j--
            if nums[j] <= x {
                break
            }
        }
        if i >= j {
            break
        }
        nums[i], nums[j] = nums[j], nums[i]
    }
    nums[j], nums[start] = nums[start], nums[j]
    return j
}
```
